## Sum Arrays

```js

function sum (numbers) {
    "use strict";
  
  let ans = 0
    
  for (let i=0; i < numbers.length; i++) {
      ans += numbers[i];
  }
  return ans     
}

```

## Reversed Strings

```js

function solution (str) {
  let reverse = '';
  for (let i = str.length; i>=0; i--) {
    reverse += str[i]
  }
  return reverse
}


```
